-- ===================== FONCTIONS CORE =====================

local function sendMsg(msg)
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local channel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if channel then channel:SendAsync(msg) end
    else
        local remote = ReplicatedStorage:FindFirstChild("SayMessageRequest", true)
        if remote then remote:FireServer(msg, "All") end
    end
end

local function executeTroll(target)
    local commands = {":rocket ", ":ballon ", ":inverse ", ":morph ", ":tiny ", ":jumpscare "}
    for _, cmd in ipairs(commands) do
        task.spawn(function()
            sendMsg(cmd .. target)
        end)
    end
end

        Instance.new("UICorner", headshot).CornerRadius = UDim.new(1, 0)

        local pLabel = Instance.new("TextLabel")
        pLabel.Name = "PlayerName"
        pLabel.Size = UDim2.new(1, -35, 1, 0)
        pLabel.Position = UDim2.new(0, 32, 0, 0)
        pLabel.BackgroundTransparency = 1

        pLabel.TextSize = 11
        pLabel.Font = Enum.Font.GothamMedium
        pLabel.TextXAlignment = Enum.TextXAlignment.Left
        pLabel.Parent = pBtn

        -- Hover Effects
        pBtn.MouseEnter:Connect(function()
            TweenService:Create(pBtn, TweenInfo.new(0.2), {BackgroundTransparency = 0.8}):Play()
            TweenService:Create(pLabel, TweenInfo.new(0.2), {TextColor3 = Color3.fromRGB(255, 50, 50)}):Play()

        pBtn.MouseLeave:Connect(function()
            TweenService:Create(pBtn, TweenInfo.new(0.2), {BackgroundTransparency = 0.93}):Play()
            TweenService:Create(pLabel, TweenInfo.new(0.2), {TextColor3 = Color3.fromRGB(240, 240, 240)}):Play()
        end)

        pBtn.MouseButton1Click:Connect(function()
            executeTroll(p.Name)
        end)
    end
    scroll.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y)
end

Players.PlayerAdded:Connect(updatePlayerList)

-- ===================== SYSTÈME DRAG (MOUVEMENT) =====================

local dragging, dragInput, dragStart, startPos

local function update(input)
	local delta = input.Position - dragStart
	main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

main.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = main.Position
		
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

main.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)
