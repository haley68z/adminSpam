-- Fonction pour envoyer un message dans le chat
local function sendMessage(content)
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local generalChannel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if generalChannel then
            generalChannel:SendAsync(content)
        end
    else
        local sayMessageRequest = ReplicatedStorage:FindFirstChild("SayMessageRequest", true)
        if sayMessageRequest then
            sayMessageRequest:FireServer(content, "All")
        end
    end
end

-- Fonction pour envoyer les 3 commandes de spam
local function spamCommands(targetName)
    local commands = {":rocket ", ":balloon ", ":morph "}
    for _, cmd in ipairs(commands) do
        task.spawn(function()
            sendMessage(cmd .. targetName)
        end)
    end
end

-- Drag System
local dragging, dragInput, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
