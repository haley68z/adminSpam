-- ===================== FONCTIONS CORE =====================

local function sendMsg(msg)
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local channel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if channel then channel:SendAsync(msg) end
    else
        local remote = ReplicatedStorage:FindFirstChild("SayMessageRequest", true)
        if remote then remote:FireServer(msg, "All") end
    end
end

local function executeTroll(target)
    local commands = {":rocket ", ":balloon ", ":jail ", ":morph "}
    for _, cmd in ipairs(commands) do
        task.spawn(function()
            sendMsg(cmd .. target)
        end)
    end
end

        headshot.BackgroundTransparency = 1
        headshot.Image = Players:GetUserThumbnailAsync(p.UserId, Enum.ThumbnailType.HeadShot, Enum.ThumbnailSize.Size48x48)
        headshot.Parent = pBtn
        Instance.new("UICorner", headshot).CornerRadius = UDim.new(1, 0) -- Rond

        local nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, -35, 1, 0)
        nameLabel.Position = UDim2.new(0, 32, 0, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = p.DisplayName or p.Name
        nameLabel.TextColor3 = Color3.fromRGB(240, 240, 240)
        nameLabel.TextSize = 11
        nameLabel.Font = Enum.Font.GothamMedium
        nameLabel.TextXAlignment = Enum.TextXAlignment.Left
        nameLabel.Parent = pBtn

        -- Hover Effects
        pBtn.MouseEnter:Connect(function()
            TweenService:Create(pBtn, TweenInfo.new(0.2), {BackgroundTransparency = 0.2, BackgroundColor3 = Color3.fromRGB(50, 20, 20)}):Play()
        end)
        pBtn.MouseLeave:Connect(function()
            TweenService:Create(pBtn, TweenInfo.new(0.2), {BackgroundTransparency = 0.5, BackgroundColor3 = Color3.fromRGB(30, 30, 30)}):Play()
        end)

        pBtn.MouseButton1Click:Connect(function()
            executeTroll(p.Name)
        end)
    end
    scroll.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y)
end

Players.PlayerAdded:Connect(updatePlayerList)

-- ===================== SYSTÈME DRAG (MOUVEMENT) =====================

local dragging, dragInput, dragStart, startPos

local function update(input)
	local delta = input.Position - dragStart
	main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

main.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = main.Position
		
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

main.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)
